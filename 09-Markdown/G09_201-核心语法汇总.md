  }, 1000);

});
```

### 2. Promise

```javascript
function fetchData() {
  return new Promise(resolve => {
    setTimeout(() => {
      resolve('');
    }, 1000);
  });
}

fetchData().then(data => {
  console.log(data);
});
```

### 3. async/await

```javascript
async function fetchData() {
  return new Promise(resolve => {
    setTimeout(() => {
      resolve('');
    }, 1000);
  });
}

async function main() {
  const data = await fetchData();
  console.log(data);
}

main();
```
