## 1. 变量与常量 (Variables & Constants)

### 1.1 变量声明

GDScript 支持动态类型和静态类型（Godot 4 推荐）。

#### 1.1.1 动态类型

```gdscript
# 动态类型变量，类型会根据赋值自动推断
var x = 10  # 整数
var y = 3.14  # 浮点数
var z = "Hello"  # 字符串
var is_active = true  # 布尔值
```

#### 1.1.2 静态类型

Godot 4 引入了静态类型系统，推荐使用以获得更好的代码提示和类型检查：

```gdscript
# 显式静态类型
var score: int = 100
var health: float = 95.5
var name: String = "Player"
var is_alive: bool = true

# 类型自动推断（使用 := 操作符）
var speed := 500  # 推断为 int
var position := Vector2(0, 0)  # 推断为 Vector2
```

#### 1.1.3 变量作用域

```gdscript
# 全局变量（在脚本顶部定义）
var global_var = "Global"

func _ready():
    # 局部变量（在函数内部定义）
    var local_var = "Local"
    print(global_var)  # 可以访问全局变量
    print(local_var)  # 可以访问局部变量

func another_function():
    print(global_var)  # 可以访问全局变量
    # print(local_var)  # 错误：局部变量不可访问
```

### 1.2 常量

使用 `const` 关键字定义常量，一旦定义不可修改：

```gdscript
# 数值常量
const GRAVITY = 9.8
const MAX_HEALTH = 100

# 字符串常量
const GAME_TITLE = "My Game"

# 向量常量
const DEFAULT_POSITION = Vector2(0, 0)

# 颜色常量
const PRIMARY_COLOR = Color(1, 0, 0)  # 红色
```

**注意**：常量必须在定义时初始化，且不能在运行时修改。

## 2. 数据类型 (Data Types)

### 2.1 基本类型

| 类型 | 描述 | 示例 |
|------|------|------|
| `int` | 整数 | `var score: int = 100` |
| `float` | 浮点数 | `var speed: float = 10.5` |
| `bool` | 布尔值 | `var is_active: bool = true` |
| `String` | 字符串 | `var name: String = "Player"` |

**字符串操作示例**：

```gdscript
var greeting = "Hello"
var name = "World"

# 字符串拼接
var message = greeting + " " + name  # "Hello World"

# 字符串格式化
var score = 100
var formatted = "Your score: %d" % score  # "Your score: 100"

# 字符串方法
var text = "   Hello World   "
print(text.trim())  # "Hello World"
print(text.to_lower())  # "   hello world   "
print(text.length())  # 13
```

### 2.2 引擎内置类型 (Built-in)

#### 2.2.1 向量类型

| 类型 | 描述 | 示例 |
|------|------|------|
| `Vector2` | 2D 向量 | `var position: Vector2 = Vector2(100, 200)` |
| `Vector3` | 3D 向量 | `var position: Vector3 = Vector3(100, 200, 50)` |
| `Vector2i` | 2D 整数向量 | `var grid_pos: Vector2i = Vector2i(5, 3)` |
| `Vector3i` | 3D 整数向量 | `var grid_pos: Vector3i = Vector3i(5, 3, 2)` |

**向量操作示例**：

```gdscript
var pos = Vector2(100, 200)
var dir = Vector2(1, 0)  # 向右
var speed = 100

# 向量加法
var new_pos = pos + dir * speed

# 向量方法
print(pos.length())  # 计算长度
print(pos.normalized())  # 归一化
print(pos.distance_to(Vector2(0, 0)))  # 计算距离
```

#### 2.2.2 其他内置类型

| 类型 | 描述 | 示例 |
|------|------|------|
| `Color` | 颜色 (RGBA) | `var color: Color = Color(1, 0, 0, 1)` |
| `Rect2` | 2D 矩形区域 | `var rect: Rect2 = Rect2(0, 0, 100, 100)` |
| `Rect2i` | 2D 整数矩形 | `var rect: Rect2i = Rect2i(0, 0, 100, 100)` |
| `Transform2D` | 2D 变换 | `var transform: Transform2D = Transform2D.IDENTITY` |
| `Transform3D` | 3D 变换 | `var transform: Transform3D = Transform3D.IDENTITY` |
| `Quaternion` | 四元数（用于 3D 旋转） | `var rotation: Quaternion = Quaternion()` |
| `AABB` | 轴对齐包围盒 | `var bounds: AABB = AABB(Vector3(0, 0, 0), Vector3(1, 1, 1))` |
| `RID` | 资源 ID | `var texture_rid: RID = texture.get_rid()` |
| `NodePath` | 节点路径 | `var path: NodePath = "Player/Sprite"` |
| `Callable` | 可调用对象 | `var func: Callable = self._on_button_pressed` |
| `Signal` | 信号 | `signal button_pressed` |

### 2.3 容器类型

#### 2.3.1 数组 (Array)

```gdscript
# 创建数组
var numbers = [1, 2, 3, 4, 5]
var names = ["Alice", "Bob", "Charlie"]
var mixed = [1, "hello", Vector2(1, 2)]

# 数组操作
numbers.append(6)  # 添加元素
numbers.remove_at(0)  # 移除第一个元素
numbers.insert(0, 0)  # 在指定位置插入元素
print(numbers.size())  # 获取数组长度

# 数组方法
var reversed = numbers.reversed()  # 反转数组
var sorted = numbers.sort()  # 排序数组
var filtered = numbers.filter(func(x): return x > 3)  # 过滤数组
```

#### 2.3.2 字典 (Dictionary)

```gdscript
# 创建字典
var player_data = {
    "name": "Player",

# 访问和修改字典
```

## 3. 控制流 (Control Flow)

### 3.1 条件判断

```gdscript
# 基本 if-elif-else
var score = 75
if score >= 90:
    print("Excellent")
elif score >= 70:
    print("Good")
elif score >= 60:
    print("Pass")
else:
    print("Fail")

# 嵌套条件
var is_admin = true
var is_logged_in = true

if is_logged_in:
    if is_admin:
        print("Welcome Admin")
    else:
        print("Welcome User")
else:
    print("Please log in")

# 单行条件表达式
var status = "Online" if is_logged_in else "Offline"
print(status)
```

### 3.2 模式匹配 (Match)

`match` 语句类似于其他语言的 switch 语句，但功能更强大：

```gdscript
# 基本匹配
var state = "idle"
match state:
    "idle":
        print("Character is idle")
    "run":
        print("Character is running")
    "jump":
        print("Character is jumping")
    _:
        print("Unknown state")

# 匹配数字范围
var score = 75
match score:
    0 <= score < 60:
        print("Fail")
    60 <= score < 70:
        print("Pass")
    70 <= score < 90:
        print("Good")
    90 <= score <= 100:
        print("Excellent")
    _:
        print("Invalid score")

# 匹配数组和字典
var data = ["player", 100]
match data:
    ["player", health]:
        print("Player health: %d" % health)
    ["enemy", health, type]:
        print("Enemy %s health: %d" % [type, health])
    _:
        print("Unknown data")

# 匹配对象类型
var obj = Vector2(1, 2)
match obj:
    Vector2:
        print("This is a Vector2")
    String:
        print("This is a String")
    _:
        print("Unknown type")
```

### 3.3 循环

#### 3.3.1 for 循环

```gdscript
# 遍历范围
for i in range(5):
    print(i)  # 输出: 0, 1, 2, 3, 4

# 遍历数组
var numbers = [1, 2, 3, 4, 5]
for number in numbers:
    print(number)

# 遍历字典
var player = {"name": "Alice", "age": 25, "score": 100}
for key in player.keys():
    print("%s: %s" % [key, player[key]])

# 遍历字典（键值对）
for key, value in player.items():
    print("%s: %s" % [key, value])

# 带索引的遍历
for i, value in enumerate(numbers):
    print("Index %d: %d" % [i, value])

# 反向遍历
for i in range(4, -1, -1):
    print(i)  # 输出: 4, 3, 2, 1, 0
```

#### 3.3.2 while 循环

```gdscript
# 基本 while 循环
var count = 0
while count < 5:
    print(count)
    count += 1

# 无限循环（需要在循环内 break）
var running = true
while running:
    var input = input_string()
    if input == "quit":
        running = false
    else:
        print("You entered: %s" % input)
```

#### 3.3.3 循环控制语句

```gdscript
# break 语句
for i in range(10):
    if i == 5:
        break  # 跳出循环
    print(i)

# continue 语句
for i in range(10):
    if i % 2 == 0:
        continue  # 跳过本次循环
    print(i)  # 只输出奇数

# 标签和 break
outer_loop:
for i in range(3):
    for j in range(3):
        if i == 1 and j == 1:
            break outer_loop  # 跳出外层循环
        print("i: %d, j: %d" % [i, j])
```

## 4. 运算符 (Operators)

### 4.1 算术运算符

| 运算符 | 描述 | 示例 |
|--------|------|------|
| `+` | 加法 | `1 + 2` → `3` |
| `-` | 减法 | `5 - 3` → `2` |
| `*` | 乘法 | `2 * 3` → `6` |
| `/` | 除法 | `10 / 2` → `5.0` |
| `//` | 整数除法 | `10 // 3` → `3` |
| `%` | 取模 | `10 % 3` → `1` |
| `**` | 幂运算 | `2 ** 3` → `8` |

### 4.2 比较运算符

| 运算符 | 描述 | 示例 |
|--------|------|------|
| `==` | 等于 | `1 == 1` → `true` |
| `!=` | 不等于 | `1 != 2` → `true` |
| `<` | 小于 | `1 < 2` → `true` |
| `<=` | 小于等于 | `1 <= 1` → `true` |
| `>` | 大于 | `2 > 1` → `true` |
| `>=` | 大于等于 | `2 >= 2` → `true` |

### 4.3 逻辑运算符

| 运算符 | 描述 | 示例 |
|--------|------|------|
| `and` | 逻辑与 | `true and false` → `false` |
| `or` | 逻辑或 | `true or false` → `true` |
| `not` | 逻辑非 | `not true` → `false` |

### 4.4 赋值运算符

| 运算符 | 描述 | 示例 |
|--------|------|------|
| `=` | 赋值 | `var x = 10` |
| `+=` | 加赋值 | `x += 5` → `x = x + 5` |
| `-=` | 减赋值 | `x -= 5` → `x = x - 5` |
| `*=` | 乘赋值 | `x *= 2` → `x = x * 2` |
| `/=` | 除赋值 | `x /= 2` → `x = x / 2` |
| `%=` | 取模赋值 | `x %= 3` → `x = x % 3` |

### 4.5 其他运算符

| 运算符 | 描述 | 示例 |
|--------|------|------|

## 5. 函数 (Functions)

### 5.1 函数定义

```gdscript
# 基本函数
func greet():
    print("Hello!")

# 带参数的函数
func greet_name(name):
    print("Hello, %s!" % name)

# 带返回值的函数
func add(a, b):
    return a + b

# 带类型注解的函数
func multiply(a: int, b: int) -> int:
    return a * b

# 带默认参数的函数
func greet_with_message(name, message = "How are you?"):
    print("Hello, %s! %s" % [name, message])

# 调用函数
greet()
greet_name("Alice")
var result = add(5, 3)
print(result)
var product = multiply(4, 6)
print(product)
greet_with_message("Bob")
greet_with_message("Charlie", "Nice to meet you!")
```

### 5.2 匿名函数 (Lambda)

```gdscript
# 基本匿名函数
var add = func(a, b):
    return a + b

print(add(2, 3))  # 输出: 5

# 作为参数传递
var numbers = [1, 2, 3, 4, 5]
var even_numbers = numbers.filter(func(x):
    return x % 2 == 0
)
print(even_numbers)  # 输出: [2, 4]

# 简写形式
var double = func(x): return x * 2
var doubled = numbers.map(double)
print(doubled)  # 输出: [2, 4, 6, 8, 10]
```

## 7. 错误处理 (Error Handling)

### 7.1 断言 (Assert)

```gdscript
# 基本断言
var value = 10
assert(value > 0, "Value must be positive")

# 带条件的断言
func divide(a, b):
    assert(b != 0, "Division by zero")
    return a / b

    print("Error occurred")
```

```gdscript
except error:
```

## 8. 代码风格与最佳实践

### 8.1 命名规范

- **变量和函数**: 使用蛇形命名法（snake_case）
  ```gdscript
  var player_health = 100
  func calculate_damage():
      pass
  ```

- **常量**: 使用全大写加下划线
  ```gdscript
  const MAX_HEALTH = 100
  const GRAVITY = 9.8
  ```

- **类名**: 使用帕斯卡命名法（PascalCase）
  ```gdscript
  class_name PlayerCharacter
  ```

### 8.2 代码组织

- 使用缩进（4 空格）
- 为复杂代码添加注释
- 合理使用空行分隔代码块
- 函数长度不宜过长，建议不超过 50 行

### 8.3 性能优化

- 使用静态类型以获得更好的性能
- 避免在循环中创建新对象
- 使用 `preload()` 预加载资源
- 合理使用信号（Signal）而非轮询

### 8.4 常见错误与解决方案

| 错误 | 原因 | 解决方案 |
|------|------|----------|
| **类型错误** | 变量类型不匹配 | 使用静态类型注解或类型检查 |
| **空引用错误** | 访问了 null 对象的属性 | 使用 `if obj != null` 检查 |
| **数组越界** | 访问了不存在的数组索引 | 使用 `if index >= 0 and index < array.size()` 检查 |
| **字典键不存在** | 访问了不存在的字典键 | 使用 `if dict.has(key)` 检查 |
| **函数参数错误** | 传递了错误类型的参数 | 使用类型注解并检查参数类型 |

## 9. 实战示例

### 9.1 玩家控制器

```gdscript
# PlayerController.gd
class_name PlayerController

var speed: float = 300.0
var jump_force: float = 500.0
var is_grounded: bool = false

func _process(delta: float):
    var velocity = Vector2.ZERO
    
    # 移动输入
    if Input.is_action_pressed("ui_right"):
        velocity.x += speed
    if Input.is_action_pressed("ui_left"):
        velocity.x -= speed
    
    # 跳跃
    if Input.is_action_just_pressed("ui_accept") and is_grounded:
        velocity.y = -jump_force
    
    # 应用重力
    velocity.y += 9.8 * 100 * delta
    
    # 移动玩家
    move_and_slide(velocity, Vector2.UP)

func _on_grounded_body_entered(body):
    is_grounded = true

func _on_grounded_body_exited(body):
    is_grounded = false
```

### 9.2 简单的敌人 AI

```gdscript
# EnemyAI.gd
class_name EnemyAI

var patrol_points = [Vector2(0, 0), Vector2(100, 0), Vector2(100, 100), Vector2(0, 100)]
var current_point = 0
var speed = 100

func _process(delta: float):
    var target = patrol_points[current_point]
    var direction = (target - global_position).normalized()
    var velocity = direction * speed
    
    move_and_slide(velocity)
    
    # 到达目标点
    if global_position.distance_to(target) < 10:
        current_point = (current_point + 1) % patrol_points.size()
```

### 9.3 计分系统

```gdscript
# ScoreManager.gd
class_name ScoreManager

var score: int = 0
var high_score: int = 0

func _ready():
    # 加载最高分
    high_score = load_high_score()

func add_score(points: int):
    score += points
    update_ui()
    
    # 更新最高分
    if score > high_score:
        high_score = score
        save_high_score()

func reset_score():
    score = 0
    update_ui()

func load_high_score() -> int:
    var config = ConfigFile.new()
    var error = config.load("user://scores.cfg")
    if error == OK:
        return config.get_value("scores", "high_score", 0)
    return 0

func save_high_score():
    var config = ConfigFile.new()
    config.set_value("scores", "high_score", high_score)
    config.save("user://scores.cfg")

func update_ui():
    # 更新 UI 显示
    var score_label = get_node("UI/ScoreLabel")
    var high_score_label = get_node("UI/HighScoreLabel")
    
    if score_label:
        score_label.text = "Score: %d" % score
    if high_score_label:
        high_score_label.text = "High Score: %d" % high_score
```
