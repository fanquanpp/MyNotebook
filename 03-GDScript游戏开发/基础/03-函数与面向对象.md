## 1. 函数

### 1.1 函数定义

GDScript 中函数使用 `func` 关键字定义：

```gdscript
# 基本函数定义
func say_hello():
    print("Hello, GDScript!")

# 带参数的函数
func greet(name):
    print("Hello, " + name + "!")

# 带返回值的函数
func add(a, b):
    return a + b
```

### 1.2 参数类型

GDScript 支持类型提示，提高代码可读性和编辑器智能提示：

```gdscript
# 带类型提示的参数
func calculate_damage(attack: int, defense: int):
    return max(1, attack - defense)

# 复合类型参数
func move_to(position: Vector2):
    self.position = position

# 自定义类型参数
func add_item(item: Item):
    inventory.append(item)
```

### 1.3 返回值类型

可以指定函数的返回值类型：

```gdscript
# 带返回值类型的函数
func add(a: int, b: int) -> int:
    return a + b

# 返回复合类型
func get_position() -> Vector2:
    return position

# 返回自定义类型
func get_player() -> Player:
    return player
```

### 1.4 默认参数

函数参数可以设置默认值：

```gdscript
# 带默认参数的函数
func greet(name: String = "Guest"):
    print("Hello, " + name + "!")

# 多个默认参数
func create_enemy(name: String = "Goblin", health: int = 100, damage: int = 10):
    var enemy = Enemy.new()
    enemy.name = name
    enemy.health = health
    enemy.damage = damage
    return enemy
```

### 1.5 可变参数

使用 `...` 表示可变参数：

```gdscript
# 可变参数函数
func sum_values(...values):
    var total = 0
    for value in values:
        total += value
    return total

# 调用示例
sum_values(1, 2, 3)  # 返回 6
sum_values(10, 20, 30, 40)  # 返回 100
```

### 1.6 函数重载

GDScript 不支持传统意义上的函数重载，但可以通过默认参数和类型检查来模拟：

```gdscript
# 模拟函数重载
func get_value(value = null):
    if value == null:
        return 0
    elif typeof(value) == TYPE_INT:
        return value
    elif typeof(value) == TYPE_STRING:
        return int(value)
    else:
        return 0
```

### 1.7 递归函数

函数可以递归调用自身：

```gdscript
# 递归函数：计算阶乘
func factorial(n: int) -> int:
    if n <= 1:
        return 1
    else:
        return n * factorial(n - 1)

# 递归函数：斐波那契数列
func fibonacci(n: int) -> int:
    if n <= 1:
        return n
    else:
        return fibonacci(n - 1) + fibonacci(n - 2)
```

### 1.8 内联函数

使用 `func` 关键字可以在函数内部定义内联函数：

```gdscript
func process_items(items):
    # 内联函数
    func process_item(item):
        print("Processing item: " + item.name)
        return item.value * 2
    
    var results = []
    for item in items:
        results.append(process_item(item))
    return results
```

## 2. 类与对象

### 2.1 类的定义

在 GDScript 中，每个 `.gd` 文件本质上都是一个类。可以使用 `class_name` 为类指定一个名称，方便在其他地方直接使用：

```gdscript
# 定义一个名为 Player 的类，继承自 CharacterBody2D
class_name Player extends CharacterBody2D

# 类的成员变量
var health: int = 100
var speed: float = 300.0
var weapon: Weapon = null

# 类的方法
func attack():
    if weapon:
        weapon.use()

func take_damage(amount: int):
    health -= amount
    if health <= 0:
        die()

func die():
    print("Player died")
```

### 2.2 成员变量

类的成员变量可以在类定义中声明：

```gdscript
class_name Enemy extends CharacterBody2D

# 公共成员变量
var name: String = "Enemy"
var health: int = 100

# 私有成员变量（以下划线开头）
var _speed: float = 100.0
var _attack_cooldown: float = 0.0

# 导出变量（在编辑器中可见和修改）
export var max_health: int = 100
export var move_speed: float = 100.0
export var damage: int = 10
```

### 2.3 构造函数

使用 `_init()` 方法作为构造函数：

```gdscript
class_name Item

var name: String
var value: int
var weight: float

# 构造函数
func _init(name: String, value: int, weight: float = 1.0):
    self.name = name
    self.value = value
    self.weight = weight
    print("Item created: " + name)

# 调用示例
var potion = Item.new("Health Potion", 50, 0.5)
var sword = Item.new("Iron Sword", 100)
```

### 2.4 析构函数

使用 `_exit_tree()` 或 `_finalize()` 方法作为析构函数：

```gdscript
class_name Resource

var name: String

func _init(name: String):
    self.name = name
    print("Resource created: " + name)

func _exit_tree():
    print("Resource exiting tree: " + name)
    # 清理资源

func _finalize():
    print("Resource finalized: " + name)
    # 最终清理
```

### 2.5 实例化对象

使用 `new()` 方法实例化对象：

```gdscript
# 实例化 Player 对象
var player = Player.new()
player.health = 100
player.speed = 300.0

# 实例化 Enemy 对象
var enemy = Enemy.new()
enemy.name = "Goblin"
enemy.health = 50

# 将对象添加到场景树
add_child(player)
add_child(enemy)
```

### 2.6 访问控制

GDScript 使用命名约定来控制访问权限：

- **公共成员**：直接使用变量名，如 `health`
- **私有成员**：以下划线开头，如 `_speed`

```gdscript
class_name Player

var public_health = 100  # 公共成员
var _private_health = 100  # 私有成员

func get_health() -> int:
    return _private_health

func set_health(value: int):
    _private_health = max(0, value)
```

## 3. 继承与多态

### 3.1 继承的基本概念

GDScript 支持类的继承，使用 `extends` 关键字：

```gdscript
# 基类
class_name Character extends Node2D

var health: int = 100
var speed: float = 100.0

func move(direction: Vector2):
    position += direction * speed

func take_damage(amount: int):
    health -= amount
    if health <= 0:
        die()

func die():
    print("Character died")

# 派生类
class_name Player extends Character

var weapon: Weapon = null

# 重写父类方法
func attack():
    if weapon:
        weapon.use()

# 扩展父类方法
func move(direction: Vector2) -> void:
    # 调用父类的 move 方法
    .move(direction)
    # 添加额外的逻辑
    print("Player moved")
```

### 3.2 方法重写

派生类可以重写父类的方法：

```gdscript
class_name Enemy extends Character

var damage: int = 10

# 重写父类的 die 方法
func die():
    print("Enemy died")
    # 掉落物品
    drop_loot()

func drop_loot():
    var loot = Item.new("Gold", 50)
    # 添加到场景
    get_parent().add_child(loot)
    loot.position = position
```

### 3.3 多态

多态允许不同类型的对象对相同的方法调用做出不同的响应：

```gdscript
# 多态示例
func damage_character(character: Character):
    character.take_damage(20)
    character.die()  # 调用的是具体子类的 die 方法

# 调用示例
var player = Player.new()
var enemy = Enemy.new()

damage_character(player)  # 调用 Player 的 take_damage 和 die 方法
damage_character(enemy)   # 调用 Enemy 的 take_damage 和 die 方法
```

### 3.4 抽象类

GDScript 没有内置的抽象类支持，但可以通过约定来实现：

```gdscript
# 抽象基类
class_name Character

func attack():
    # 抽象方法，子类必须实现
    push_error("Subclass must implement attack()")

func move():
    # 抽象方法，子类必须实现
    push_error("Subclass must implement move()")

# 具体子类
class_name Player extends Character

func attack():
    print("Player attacks")

func move():
    print("Player moves")
```

### 3.5 接口

GDScript 没有内置的接口支持，但可以通过多重继承或组合来实现：

```gdscript
# 接口类
class_name Damageable

func take_damage(amount: int):
    push_error("Subclass must implement take_damage()")

func die():
    push_error("Subclass must implement die()")

# 实现接口
class_name Player extends Character, Damageable

func take_damage(amount: int):
    health -= amount
    if health <= 0:
        die()

func die():
    print("Player died")
```

## 4. 静态成员

### 4.1 静态变量

静态变量属于类而不是实例，使用 `static` 关键字：

```gdscript
class_name GameManager

# 静态变量
static var score: int = 0
static var lives: int = 3
static var game_over: bool = false

# 静态方法
static func add_score(points: int):
    score += points
    print("Score: " + str(score))

static func lose_life():
    lives -= 1
    if lives <= 0:
        game_over = true
        print("Game Over!")

# 调用示例
GameManager.add_score(100)
GameManager.lose_life()
```

### 4.2 静态方法

静态方法不依赖于实例，直接通过类名调用：

```gdscript
class_name MathUtils

# 静态方法
static func clamp(value: float, min: float, max: float) -> float:
    return max(min, min(value, max))

static func lerp(a: float, b: float, t: float) -> float:
    return a + (b - a) * t

static func random_range(min: float, max: float) -> float:
    return randf_range(min, max)

# 调用示例
var clamped_value = MathUtils.clamp(15, 0, 10)  # 返回 10
var lerped_value = MathUtils.lerp(0, 100, 0.5)  # 返回 50
var random_value = MathUtils.random_range(1, 10)  # 返回 1-10 之间的随机数
```

## 5. Lambda 表达式

### 5.1 基本语法

Lambda 表达式是一种匿名函数，使用 `func` 关键字定义：

```gdscript
# 基本 Lambda 表达式
var square = func(x): return x * x
print(square.call(5))  # 输出 25

# 带类型提示的 Lambda 表达式
var add = func(a: int, b: int) -> int: return a + b
print(add.call(3, 4))  # 输出 7

# 多行 Lambda 表达式
var process = func(value):
    var result = value * 2
    result += 10
    return result
print(process.call(5))  # 输出 20
```

### 5.2 使用场景

Lambda 表达式常用于临时回调、事件处理等场景：

```gdscript
# 作为参数传递
func process_array(array, processor):
    var result = []
    for item in array:
        result.append(processor.call(item))
    return result

# 使用 Lambda 表达式作为参数
var numbers = [1, 2, 3, 4, 5]
var squared = process_array(numbers, func(x): return x * x)  # 返回 [1, 4, 9, 16, 25]

# 事件处理
var button = Button.new()
button.pressed.connect(func(): print("Button pressed"))

# 排序
var players = [
    {name: "Alice", score: 100},
    {name: "Bob", score: 200},
    {name: "Charlie", score: 50}
]
players.sort_custom(func(a, b): return a.score > b.score)  # 按分数降序排序
```

## 6. 信号与事件

### 6.1 信号的定义

信号是 Godot 中用于节点间通信的机制，使用 `signal` 关键字定义：

```gdscript
class_name Player extends CharacterBody2D

# 定义信号
signal health_changed(health)
signal died()
signal attacked(damage)

var health: int = 100

func take_damage(amount: int):
    health -= amount
    # 发射信号
    health_changed.emit(health)
    if health <= 0:
        died.emit()

func attack():
    # 发射信号
    attacked.emit(20)
```

### 6.2 信号的连接

使用 `connect()` 方法连接信号到处理函数：

```gdscript
# 连接信号
var player = Player.new()

# 连接到普通函数
player.health_changed.connect(self._on_health_changed)
player.died.connect(self._on_player_died)

# 连接到 Lambda 表达式
player.attacked.connect(func(damage): print("Player attacked for " + str(damage) + " damage"))

# 处理函数
func _on_health_changed(health):
    print("Player health: " + str(health))

func _on_player_died():
    print("Player died!")
```

### 6.3 信号的发射

使用 `emit()` 方法发射信号：

```gdscript
class_name Button extends Control

# 定义信号
signal pressed()
signal released()

func _input(event):
    if event.is_action_pressed("ui_accept"):
        # 发射 pressed 信号
        pressed.emit()
    elif event.is_action_released("ui_accept"):
        # 发射 released 信号
        released.emit()
```

### 6.4 信号与 Lambda 表达式

Lambda 表达式常用于临时处理信号：

```gdscript
var button = Button.new()

# 使用 Lambda 表达式处理信号
button.pressed.connect(func():
    print("Button pressed!")
    # 执行其他逻辑
)

# 带参数的 Lambda 表达式
var slider = HSlider.new()
slider.value_changed.connect(func(value):
    print("Slider value: " + str(value))
)
```

## 7. 设计模式

### 7.1 单例模式

单例模式确保一个类只有一个实例：

```gdscript
# 单例类
class_name GameManager

static var _instance: GameManager = null

# 获取单例实例
static func get_instance() -> GameManager:
    if _instance == null:
        _instance = GameManager.new()
    return _instance

# 成员变量
var score: int = 0
var lives: int = 3

# 方法
func add_score(points: int):
    score += points

func lose_life():
    lives -= 1

# 调用示例
var game_manager = GameManager.get_instance()
game_manager.add_score(100)
```

### 7.2 工厂模式

工厂模式用于创建对象，隐藏创建细节：

```gdscript
# 工厂类
class_name EnemyFactory

static func create_enemy(enemy_type: String) -> Enemy:
    var enemy = null
    
    match enemy_type:
        "goblin":
            enemy = Goblin.new()
        "orc":
            enemy = Orc.new()
        "troll":
            enemy = Troll.new()
        _:
            enemy = Goblin.new()  # 默认敌人
    
    return enemy

# 调用示例
var goblin = EnemyFactory.create_enemy("goblin")
var orc = EnemyFactory.create_enemy("orc")
```

### 7.3 观察者模式

观察者模式通过信号机制实现：

```gdscript
# 主题类
class_name Subject

signal state_changed(new_state)

var state: String = "idle"

func set_state(new_state: String):
    state = new_state
    state_changed.emit(new_state)

# 观察者类
class_name Observer

func _init(subject: Subject):
    # 连接信号
    subject.state_changed.connect(self._on_state_changed)

func _on_state_changed(new_state):
    print("State changed to: " + new_state)

# 调用示例
var subject = Subject.new()
var observer = Observer.new(subject)

subject.set_state("active")  # 输出 "State changed to: active"
subject.set_state("inactive")  # 输出 "State changed to: inactive"
```

## 8. 最佳实践

### 8.1 函数设计

- **函数职责单一**：每个函数只负责一项具体任务
- **函数命名清晰**：使用描述性的函数名，如 `calculate_damage` 而不是 `calc_dmg`
- **参数数量适中**：函数参数不宜过多，一般不超过 5 个
- **返回值明确**：明确函数的返回值类型和意义
- **错误处理**：对可能的错误情况进行处理

### 8.2 类设计

- **类职责单一**：每个类只负责一项具体功能
- **类命名清晰**：使用描述性的类名，如 `PlayerCharacter` 而不是 `PC`
- **成员变量合理**：合理组织成员变量，使用私有成员变量保护内部状态
- **方法设计合理**：设计清晰、简洁的方法，避免过于复杂的方法
- **继承层次合理**：避免过深的继承层次，一般不超过 3 层

### 8.3 继承与组合

- **优先使用组合**：当类之间的关系不是 "is-a" 关系时，优先使用组合而不是继承
- **合理使用继承**：当类之间存在明确的 "is-a" 关系时，使用继承
- **避免多重继承**：GDScript 支持多重继承，但应谨慎使用，避免复杂性
- **使用接口**：通过约定实现接口，提高代码的灵活性和可扩展性

## 9. 实际应用示例

### 9.1 示例 1：角色系统

```gdscript
# 基类
class_name Character extends Node2D

var health: int = 100
var speed: float = 100.0

signal health_changed(health)
signal died()

func take_damage(amount: int):
    health -= amount
    health_changed.emit(health)
    if health <= 0:
        die()

func die():
    died.emit()
    queue_free()

# 玩家类
class_name Player extends Character

var weapon: Weapon = null
var inventory: Array[Item] = []

func attack():
    if weapon:
        weapon.use()

func add_item(item: Item):
    inventory.append(item)

func use_item(index: int):
    if index >= 0 and index < inventory.size():
        var item = inventory[index]
        item.use(self)
        inventory.remove(index)

# 敌人类
class_name Enemy extends Character

var damage: int = 10
var loot: Array[Item] = []

func _ready():
    # 初始化掉落物品
    loot.append(Item.new("Gold", 50))

func attack(target: Character):
    target.take_damage(damage)

func die():
    # 掉落物品
    for item in loot:
        var item_instance = item.duplicate()
        get_parent().add_child(item_instance)
        item_instance.position = position
    
    # 调用父类的 die 方法
    .die()
```

```gdscript
class_name UIManager extends Node

static var instance: UIManager = null

@onready var health_bar = $HealthBar
```

### 9.3 示例 3：游戏状态管理

```gdscript
# 游戏状态枚举
class_name GameState

static var MENU = "menu"
static var PLAYING = "playing"
static var PAUSED = "paused"
static var GAME_OVER = "game_over"
```

## 10. 总结

GDScript 提供了完整的面向对象编程支持，包括：

- **函数**：支持类型提示、默认参数、可变参数、递归等特性
- **类与对象**：支持类定义、成员变量、构造函数、析构函数等
- **继承与多态**：支持类的继承、方法重写、多态等特性
- **静态成员**：支持静态变量和静态方法
- **Lambda 表达式**：支持匿名函数，用于临时回调和事件处理
- **信号与事件**：支持信号机制，用于节点间通信
- **设计模式**：可以实现常见的设计模式，如单例模式、工厂模式、观察者模式等

通过掌握这些特性，开发者可以构建结构清晰、可维护的游戏代码。GDScript 的面向对象特性与 Godot 引擎的节点系统和场景系统完美结合，为游戏开发提供了强大的工具。
