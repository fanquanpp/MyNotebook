## 1. 性能优化 (Performance)

### 1.1 代码优化

#### 静态类型

```gdscript
# 动态类型（较慢）
var speed = 100

# 静态类型（较快）
var speed: float = 100.0

# 函数参数和返回值类型
def func calculate_damage(attack: float, defense: float) -> float:
    return attack * 0.8 - defense * 0.5
```

#### 减少每帧计算

```gdscript
# 不好的做法 - 每帧都进行昂贵计算
func _process(delta):
    var distance = player.position.distance_to(enemy.position)
    if distance < 10:
        attack()

# 好的做法 - 只在需要时计算
var last_distance = 0
func _process(delta):
    if get_tree().get_frame() % 5 == 0:  # 每5帧计算一次
        last_distance = player.position.distance_to(enemy.position)
    if last_distance < 10:
        attack()
```

#### 预计算

```gdscript
# 预计算常用值
var sin_values = []
func _ready():
    # 预计算正弦值
    for i in range(360):
        sin_values.append(sin(deg_to_rad(i)))

# 使用预计算值
func get_sin(angle: int) -> float:
    return sin_values[angle % 360]
```

### 1.2 内存优化

#### 对象池

```gdscript
class_name ObjectPool

var pool: Array[Node] = []
var prefab: PackedScene
var max_size: int = 10

func initialize(prefab_scene: PackedScene, size: int):
    prefab = prefab_scene
    max_size = size
    # 预创建对象
    for i in range(size):
        var obj = prefab.instantiate()
        pool.append(obj)

func get_object() -> Node:
    if pool.size() > 0:
        return pool.pop_back()
    else:
        # 超出池大小，创建新对象
        return prefab.instantiate()

func return_object(obj: Node):
    if pool.size() < max_size:
        # 重置对象状态
        reset_object(obj)
        pool.append(obj)
    else:
        # 池已满，销毁对象
        obj.queue_free()

func reset_object(obj: Node):
    # 重置对象状态的逻辑
    pass
```

#### 类型数组

```gdscript
# 普通数组（较慢）
var positions = [Vector3(0, 0, 0), Vector3(1, 0, 0), Vector3(2, 0, 0)]

# 类型数组（较快）
var packed_positions = PackedVector3Array()
packed_positions.append(Vector3(0, 0, 0))
packed_positions.append(Vector3(1, 0, 0))
packed_positions.append(Vector3(2, 0, 0))
```

## 2. 内存管理 (Memory)

### 2.1 引用计数

```gdscript
# RefCounted 对象会自动管理内存
var resource = Resource.new()
# 当没有引用时，资源会被自动释放

# 手动增加引用计数
resource.reference()
# 手动减少引用计数
resource.unreference()
```

### 2.2 节点管理

```gdscript
# 安全销毁节点
func destroy_node(node: Node):
    if is_instance_valid(node):
        node.queue_free()

# 避免直接使用 free()
# node.free()  # 不安全，可能导致崩溃
```

### 2.3 循环引用

```gdscript
# 循环引用会导致内存泄漏
var node1 = Node.new()
var node2 = Node.new()
node1.set_meta("ref", node2)
node2.set_meta("ref", node1)

# 使用弱引用避免循环引用
var node1 = Node.new()
var node2 = Node.new()
node1.set_meta("ref", weakref(node2))
node2.set_meta("ref", weakref(node1))

# 获取弱引用指向的对象
var ref = node1.get_meta("ref")
var node = ref.get_ref()
if node:
    print("Node is still valid")
else:
    print("Node has been freed")
```

### 2.4 资源管理

```gdscript
# 预加载资源
var texture = preload("res://textures/player.png")

# 动态加载资源
func load_resource(path: String):
    var resource = load(path)
    if resource:
        return resource
    else:
        print("Failed to load resource: ", path)
        return null

# 卸载未使用的资源
func unload_unused_resources():
    ResourceLoader.cleanup_unused_resources()
```

### 2.5 内存泄漏检测

```gdscript
# 定期检查内存使用情况
func check_memory_usage():
    var memory = OS.get_system_memory()
    print("System memory: ", memory["available"], "/", memory["total"])
    
    var used = memory["total"] - memory["available"]
    print("Used memory: ", used)
```

## 4. 版本差异 (Godot 3 vs 4)

### 4.1 语法差异

| 特性 | Godot 3 | Godot 4 |
|------|---------|---------|
| 信号连接 | `signal.connect(handler)` | `signal.connect(handler)` (语法相同，但底层实现不同) |
| 异步处理 | `yield(get_tree().create_timer(1.0), "timeout")` | `await get_tree().create_timer(1.0).timeout` |
| 类定义 | `class_name Player` (可选) | `class_name Player` (推荐) |
| 类型系统 | 可选的静态类型 | 增强的静态类型系统 |
| 输入处理 | `Input.is_action_pressed("ui_accept")` | `Input.is_action_pressed("ui_accept")` (相同) |
| 节点创建 | `var node = Node.new()` | `var node = Node.new()` (相同) |
| 场景实例化 | `var instance = prefab.instance()` | `var instance = prefab.instantiate()` |

### 4.2 API 变更

#### 节点系统

```gdscript
# Godot 3
var node = Node2D.new()
node.position = Vector2(100, 100)

# Godot 4
var node = Node2D.new()
node.position = Vector2(100, 100)  # 相同
```

#### 物理系统

```gdscript
# Godot 3
var body = RigidBody2D.new()
body.linear_velocity = Vector2(100, 0)

# Godot 4
var body = RigidBody2D.new()
body.linear_velocity = Vector2(100, 0)  # 相同
```

#### 渲染系统

```gdscript
# Godot 3
var sprite = Sprite.new()
sprite.texture = preload("res://textures/player.png")

# Godot 4
var sprite = Sprite2D.new()  # 注意名称变更
sprite.texture = preload("res://textures/player.png")
```

### 4.3 性能差异

| 特性 | Godot 3 | Godot 4 |
|------|---------|---------|
| 静态类型 | 支持但性能提升有限 | 显著的性能提升 |
| 垃圾回收 | 基于引用计数 | 基于引用计数 + 增量式垃圾回收 |
| 渲染性能 | OpenGL 3.3 | Vulkan / OpenGL ES 3.0 |
| 脚本执行 | GDScript 1.0 | GDScript 2.0 (更快) |
| 物理引擎 | Bullet | Godot Physics / Bullet |

### 4.4 迁移指南

#### 从 Godot 3 迁移到 Godot 4

1. **更新节点名称**：
   - `Sprite` → `Sprite2D`
   - `AnimatedSprite` → `AnimatedSprite2D`
   - `Position2D` → `Marker2D`

2. **更新异步代码**：
   ```gdscript
   # Godot 3
   yield(get_tree().create_timer(1.0), "timeout")
   
   # Godot 4
   await get_tree().create_timer(1.0).timeout
   ```

3. **更新场景实例化**：
   ```gdscript
   # Godot 3
   var instance = prefab.instance()
   
   # Godot 4
   var instance = prefab.instantiate()
   ```

4. **更新类型注解**：
   ```gdscript
   # Godot 3
   var speed = 100  # 动态类型
   
   # Godot 4
   var speed: float = 100.0  # 静态类型
   ```

## 5. 实际应用案例

### 5.1 性能优化案例

#### 子弹系统优化

```gdscript
class_name BulletSystem extends Node

@export var bullet_prefab: PackedScene
@export var max_bullets: int = 50

var bullet_pool: Array[Node2D] = []

func _ready():
    # 预创建子弹
    for i in range(max_bullets):
        var bullet = bullet_prefab.instantiate()
        bullet.visible = false
        bullet_pool.append(bullet)
        add_child(bullet)

func fire_bullet(position: Vector2, direction: Vector2):
    var bullet = get_bullet()
    if bullet:
        bullet.position = position
        bullet.direction = direction
        bullet.visible = true
        bullet.active = true

func get_bullet() -> Node2D:
    # 从池中获取子弹
    for bullet in bullet_pool:
        if not bullet.active:
            return bullet
    
    # 池已满，创建新子弹
    var bullet = bullet_prefab.instantiate()
    bullet.visible = false
    bullet_pool.append(bullet)
    add_child(bullet)
    return bullet

func return_bullet(bullet: Node2D):
    bullet.visible = false
    bullet.active = false
```

### 5.2 内存管理案例

#### 资源加载管理器

```gdscript
class_name ResourceManager extends Node

var loaded_resources: Dictionary = {}

func load_resource(path: String) -> Resource:
    if loaded_resources.has(path):
        # 资源已加载，返回引用
        return loaded_resources[path]
    else:
        # 加载新资源
        var resource = load(path)
        if resource:
            loaded_resources[path] = resource
        return resource

func unload_resource(path: String):
    if loaded_resources.has(path):
        loaded_resources.erase(path)

func unload_all_resources():
    loaded_resources.clear()
    ResourceLoader.cleanup_unused_resources()
```

### 5.3 元编程案例

#### 动态配置系统

```gdscript
class_name ConfigManager extends Node

var config: Dictionary = {}

func load_config(path: String):
    var file = FileAccess.open(path, FileAccess.READ)
    if file:
        var content = file.get_as_text()
        config = parse_json(content)
        file.close()

func get_config_value(key: String, default = null):
    var keys = key.split(".")
    var current = config
    
    for k in keys:
        if not current.has(k):
            return default
        current = current[k]
    
    return current

func set_config_value(key: String, value):
    var keys = key.split(".")
    var current = config
    
    for i in range(keys.size() - 1):
        var k = keys[i]
        if not current.has(k):
            current[k] = {}
        current = current[k]
    
    current[keys[-1]] = value

func save_config(path: String):
    var file = FileAccess.open(path, FileAccess.WRITE)
    if file:
        file.store_string(json_stringify(config))
        file.close()
```

## 6. 常见问题与解决方案

### 6.1 性能问题

**问题**：游戏运行卡顿

**解决方案**：
1. 使用性能分析器识别瓶颈
2. 优化 `_process` 和 `_physics_process` 中的代码
3. 使用对象池减少节点创建/销毁开销
4. 使用类型数组替代普通数组
5. 减少每帧的计算量，使用预计算

### 6.2 内存问题

**问题**：内存使用持续增长

**解决方案**：
1. 检查是否存在循环引用
2. 使用弱引用避免循环引用
3. 及时释放不再使用的资源
4. 定期调用 `ResourceLoader.cleanup_unused_resources()`
5. 监控内存使用情况

### 6.3 版本兼容性问题

**问题**：从 Godot 3 迁移到 Godot 4 后代码无法运行

**解决方案**：
1. 更新节点名称（如 `Sprite` → `Sprite2D`）
2. 更新异步代码（`yield` → `await`）
3. 更新场景实例化（`instance()` → `instantiate()`）
4. 检查 API 变更
5. 重新编译自定义插件

### 6.4 元编程问题

**问题**：动态访问属性或方法时出错

**解决方案**：
1. 使用 `has()` 检查属性是否存在
2. 使用 `has_method()` 检查方法是否存在
3. 使用 try-except 捕获可能的错误
4. 确保对象在访问时仍然有效

## 7. 最佳实践

### 7.1 性能最佳实践

1. **使用静态类型**：为变量、函数参数和返回值添加类型注解
2. **减少每帧计算**：避免在 `_process` 中进行昂贵的计算
3. **使用对象池**：复用频繁创建/销毁的对象
4. **使用类型数组**：对于大量相同类型的数据使用 `PackedArray`
5. **预计算**：提前计算常用值
6. **批处理渲染**：使用 `MultiMesh` 批量渲染相同的对象
7. **LOD**：根据距离使用不同细节的模型
8. **使用性能分析器**：定期分析性能瓶颈

### 7.2 内存管理最佳实践

1. **使用引用计数**：理解 Godot 的内存管理机制
2. **安全销毁节点**：使用 `queue_free()` 而不是 `free()`
3. **避免循环引用**：使用弱引用处理循环依赖
4. **管理资源**：合理加载和释放资源
5. **监控内存**：定期检查内存使用情况
6. **清理未使用资源**：使用 `ResourceLoader.cleanup_unused_resources()`

### 7.3 元编程最佳实践

1. **谨慎使用**：只在必要时使用动态访问
2. **错误处理**：添加适当的错误处理
3. **性能考虑**：动态访问比直接访问慢
4. **代码可读性**：确保代码易于理解和维护
5. **文档**：为动态访问的代码添加注释

### 7.4 版本迁移最佳实践

1. **逐步迁移**：分模块迁移代码
2. **测试**：在迁移后进行充分测试
3. **文档**：记录版本差异和迁移步骤
4. **兼容性**：考虑向后兼容性
5. **学习新特性**：充分利用新版本的新特性
